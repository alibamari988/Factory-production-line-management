# نمودار توالی اول ( ثبت سفارش مواد اولیه )
@startuml

actor "مدیر تولید"

participant "سیستم"
participant "انبار"
participant "تأمین‌کننده"

مدیر تولید -> سیستم : بررسی موجودی

سیستم -> انبار : دریافت موجودی

انبار --> سیستم : مقدار موجودی

سیستم --> مدیر تولید : کمبود مواد اولیه

مدیر تولید -> سیستم : ثبت سفارش

سیستم -> تأمین‌کننده : ارسال سفارش

تأمین‌کننده --> سیستم : تأیید سفارش

سیستم --> مدیر تولید : سفارش ثبت شد

@enduml
---
# نمودار توالی دوم ( ثبت فروش محصول )
@startuml

actor "مسئول فروش"

participant "سیستم فروش"
participant "انبار"

مسئول فروش -> سیستم فروش : ثبت فروش

سیستم فروش -> انبار : بررسی موجودی

انبار --> سیستم فروش : موجودی کافی

سیستم فروش -> انبار : کسر موجودی

انبار --> سیستم فروش : تأیید

سیستم فروش --> مسئول فروش : فاکتور فروش

@enduml
ـ---
# نمودار توالی سوم ( اختصاص محصول به خط تولید )
@startuml

actor "مدیر تولید"

participant "سیستم"
participant "خط تولید"
participant "انبار"

مدیر تولید -> سیستم : انتخاب محصول

سیستم -> انبار : بررسی مواد اولیه

انبار --> سیستم : موجودی کافی

سیستم -> خط تولید : تخصیص محصول

خط تولید --> سیستم : شروع تولید

سیستم --> مدیر تولید : برنامه تولید ثبت شد

@enduml
